# score of parentheses(865)
**ideaï¼šdivide and conquer**
``` cpp
class Solution {
public:
    int scoreOfParentheses(string S) {
        
        return divide(0, S.size(), S);
    }
    
    int divide(int start, int end, string S)
    {
        int ans = 0;
        int bal = 0;
        for (int i = start; i < end; i++)
        {
            bal += S[i] == '(' ? 1 : -1;
            if (bal == 0)
            {
                if (i - start == 1) ans++;
                else ans += 2 * divide(start+1, i, S);
                start = i+1;
            }
        }
        return ans;
    }
};
```

```
class Solution {
public:
    int scoreOfParentheses(string S) {
        
        stack<char> s;
        int cur = 0;
        for (auto ch: S){
            if (ch == '(') {
                s.push(cur); 
                cur = 0;
            }
            else {
                cur = s.top() + max(2*cur, 1);
                s.pop();
            }
        }
        return cur;
    }

};
```
