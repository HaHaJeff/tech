# find all duplicate in an array
**思路：将当前num当作下标，并将该num下标对应的值变为负数，如果再次遍历同样的num，那么如果该num下标对应的值为负，则说明重复**
``` cpp
class Solution {
public:
    vector<int> findDuplicates(vector<int>& nums) {
        vector<int> ans;
        for (auto& n : nums) {
            nums[abs(n)-1] = -nums[abs(n)-1];
            if (nums[abs(n)-1] > 0) ans.push_back(abs(n));
        }
        return ans;
    }
};
```
